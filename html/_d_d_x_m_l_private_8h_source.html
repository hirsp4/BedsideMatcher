133 @interface DDXMLAttributeNode : DDXMLNode

135  // The xmlAttrPtr type doesn't allow for ownership of a namespace.

137  // In other types, such as xmlNodePtr:
138  // - nsDef stores namespaces that are owned by the node (have been alloced by the node).
